```swift {tabTitle:Swift}
import Sentry

do {
    try aMethodThatMightFail()
} catch {
    SentrySDK.capture(error: error)
}
```

```objc {tabTitle:Objective-C}
@import Sentry;

@import Sentry;

NSError *error = nil;
[self aMethodThatMightFail:&error]

if (error) {
    [SentrySDK captureError:error];
}
```
