```elixir
# sentry.ex
defmodule MyApp.Sentry do
  def before_send(%{user: user} = event) do
    # Don't send user's email address
    user = user && Map.delete(user, :email)
    %{event | user: user}
  end
end

# config.exs
config :sentry,
  before_send_event: {MyApp.Sentry, :before_send},
  # ...
```
