In addition to errors Sentry also captures all of Unity's `Debug` messages.

```csharp
using Sentry;
using UnityEngine;

public class TestMonoBehaviour : MonoBehaviour
{
    private GameObject testObject = null;

    void Start()
    {
        // This will throw a Null Reference Exception
        testObject.GetComponent<Transform>();

        SentrySdk.CaptureMessage("Verified!");

        Debug.Log("Captured Log");
        Debug.LogWarning("Captured Warning");
        Debug.LogError("Captured Error");
    }
}
```
