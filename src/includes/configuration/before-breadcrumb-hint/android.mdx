import io.sentry.android.core.SentryAndroid;

SentryAndroid.init(options -> {
    options.setBeforeBreadcrumb((breadcrumb, hint) -> {
        if (hint instanceof MyHint) {
          return null;
        } else {
          return breadcrumb;
        }
    });
});
