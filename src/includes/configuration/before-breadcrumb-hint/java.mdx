```java {tabTitle: Java}
import io.sentry.Sentry;

Sentry.init(options -> {
    options.setBeforeBreadcrumb((breadcrumb, hint) -> {
        if (hint instanceof MyHint) {
          return null;
        } else {
          return breadcrumb;
        }
    });
});
```

```kotlin {tabTitle: Kotlin}
import io.sentry.Sentry
import io.sentry.SentryOptions.BeforeBreadcrumb

Sentry.init {
    it.beforeBreadcrumb = BeforeBreadcrumb { breadcrumb, hint ->
        if (hint is MyHint) {
            null
        } else {
            breadcrumb
        }
    }
}
```
