```java
SentryAndroid.init(this, options -> {
   if (!BuildConfig.DEBUG) {

     // default values:
     // minEventLevel = ERROR
     // minBreadcrumbLevel = INFO
     options.addIntegration(

         )
     );
   }
});
```

```kotlin
SentryAndroid.init(this) { options ->
  if (!BuildConfig.DEBUG) {

    // default values:
    // minEventLevel = ERROR
    // minBreadcrumbLevel = INFO
    options.addIntegration(
      SentryTimberIntegration(
        minEventLevel = SentryLevel.ERROR,
        minBreadcrumbLevel = SentryLevel.INFO
      )
    )
  }
}
```
