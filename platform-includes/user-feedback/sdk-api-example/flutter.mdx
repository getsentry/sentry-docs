// Option 1: Retrieving SentryId from beforeSend
SentryId sentryId = SentryId.empty();

  options.beforeSend = (event, hint) {
    sentryId = event.eventId;
    return event;
  };
});

final feedback = SentryFeedback(
    message: 'Hello World!',
    contactEmail: 'foo@bar.org',
    name: 'John Doe',
