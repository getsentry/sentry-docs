<SplitSection>
<SplitSectionText>

Opt in by adding `spanStreamingIntegration()` to your list of integrations when initializing the SDK in your `router.tsx` file:

</SplitSectionText>
<SplitSectionCode>

```javascript {filename: src/router.tsx}
import * as Sentry from "@sentry/tanstackstart-react";
import { createRouter } from "@tanstack/react-router";

export const getRouter = () => {
  const router = createRouter();

  if (!router.isServer) {
    Sentry.init({
      dsn: "___PUBLIC_DSN___",
      tracesSampleRate: 1.0,
      integrations: [
        Sentry.tanstackRouterBrowserTracingIntegration(router),
        // enables stream mode
        Sentry.spanStreamingIntegration(),
      ],
    });
  }
  return router;
};
```

</SplitSectionCode>
</SplitSection>

<SplitSection>
<SplitSectionText>

Then set the `traceLifecycle` option to `'stream'` when initializing the SDK in your `instrument.server.mjs` file:

</SplitSectionText>
<SplitSectionCode>

```javascript {filename: instrument.server.mjs}
import * as Sentry from "@sentry/tanstackstart-react";

Sentry.init({
  dsn: "___PUBLIC_DSN___",
  tracesSampleRate: 1.0,
  // enables stream mode
  traceLifecycle: "stream",
});
```

</SplitSectionCode>
</SplitSection>
</SplitLayout>

To revert to transaction mode, remove `traceLifecycle` and `spanStreamingIntegration()` from these files.
