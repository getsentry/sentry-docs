```kotlin
import io.sentry.kotlin.multiplatform.Sentry

// Trailing lambda
Sentry.logger.info("User purchased product %s", productName) {
    this["userId"] = user.id
    this["referrer"] = "google"
}

// Full DSL
Sentry.logger.info {
    message("User purchased product %s", productName)
    attributes {
        this["userId"] = user.id
        this["referrer"] = "google"
    }
}
```
