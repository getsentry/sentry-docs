```powershell
Start-Sentry {
    $_.SetBeforeBreadcrumb([System.Func[Sentry.Breadcrumb, Sentry.Breadcrumb]] {
            param([Sentry.Breadcrumb]$breadcrumb)
            # Ignore breadcrumbs from Spammy logger
            if ($breadcrumb.Category -eq "Spammy.Logger") {
                return $null
            }
            return $breadcrumb
        });
}
```
