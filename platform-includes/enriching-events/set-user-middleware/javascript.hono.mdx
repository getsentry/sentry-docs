```javascript {tabTitle:Cloudflare Workers}
import * as Sentry from "@sentry/hono/cloudflare";

app.use(async (c, next) => {
  const user = c.get("user");

  if (user) {
    Sentry.setUser({
      id: user.id,
      email: user.email,
      username: user.username,
    });
  } else {
    Sentry.setUser(null);
  }

  await next();
});
```
