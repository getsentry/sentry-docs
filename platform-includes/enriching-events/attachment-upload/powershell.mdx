```powershell
# Global Scope
Edit-SentryScope {
    [Sentry.ScopeExtensions]::AddAttachment($_, 'your/path/file.log')
}

# Local Scope
'message' | Out-Sentry -EditScope {
    [Sentry.ScopeExtensions]::AddAttachment($_, 'your/path/file.log')
}

# Clear all attachments in the global scope
Edit-SentryScope {
    $scope.ClearAttachments();
}
```
