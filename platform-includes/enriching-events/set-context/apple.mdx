```swift {tabTitle:Swift}
import Sentry

SentrySDK.configureScope { scope in
    scope.setContext(value: [
        "name": "Mighty Fighter",
        "age" : 19,
        "attack_type" : "melee"
    ], key: "character")
}
```

```objc {tabTitle:Objective-C}
@import Sentry;

[SentrySDK configureScope:^(SentryScope *_Nonnull scope) {
    [scope setContextValue:@{
        @"name" : @"Mighty Fighter",
        @"age" : @19,
        @"attack_type" : @"melee"
    } forKey:@"character"];
}];
```
