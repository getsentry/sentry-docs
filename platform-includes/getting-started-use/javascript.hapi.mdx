require("./instrument");

// Now require other modules
const Sentry = require("@sentry/node");
const Hapi = require('@hapi/hapi');

const init = async () => {
  const server = Hapi.server({
    port: 3030,
    host: 'localhost',
  });

  // All your routes etc.

  await Sentry.setupHapiErrorHandler(server);

  await server.start();

};

init();
